cmake_minimum_required(VERSION 3.13) 
set(projectname picosmsPlus)
include(pico_sdk_import.cmake) 
project(${projectname} C CXX ASM) 
set(CMAKE_C_STANDARD 11)

set(CMAKE_CXX_STANDARD 17) 
pico_sdk_init() 
pico_set_program_name(${projectname} "${projectname}")
# FOUT pico_set_program_version(${projectname} "0.1")
add_subdirectory(core)
add_subdirectory(drivers/fatfs)
add_subdirectory(drivers/sdcard)
# tinyusb
set(FAMILY rp2040)
set(BOARD pico_sdk)
set(TINYUSB_DEBUG_LEVE 0)
include(${PICO_TINYUSB_PATH}/hw/bsp/${FAMILY}/family.cmake)
# FOUT family_configure_target(${projectname})
add_executable(${projectname}
        shell/main.cpp ) 
pico_enable_stdio_usb(${projectname} 1)
pico_enable_stdio_uart(${projectname} 1)
pico_add_extra_outputs(${projectname})
target_link_libraries(${projectname} pico_stdlib smscore fatfs sdcard)
